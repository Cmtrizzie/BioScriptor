```javascript
const BLACKLIST = [
  "rm -rf", "DROP DATABASE", "eval(",
  "child_process", "System.exit"
];

export function sanitizeInput(input) {
  const threat = BLACKLIST.find(cmd => input.includes(cmd));
  if (threat) throw Error(`Blocked: ${threat}`);
  return input.trim();
}
```
